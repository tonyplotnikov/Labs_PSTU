int perem(...)
{
	int s, f, g;
	s = rand() % 50;
	f = rand() % 50;
	g = rand() % 50;
	if (s > f && s > g)
		cout << "Максимальный параметр = s = " << s << endl;
	if (f > s && f > g)
		cout << "Максимальный параметр = f = " << f << endl;
	if (g > f && g > s)
		cout << "Максимальный параметр = g = " << g << endl;
	return 0;
}
template<class T, class D>
T max(T n, D d)
{
	D* a = new D[n];
	for (int i = 0; i < n; i++)
	{
		a[i] = d;
		cout << a[i] << ' ';
	}
	cout << endl;
	D k;
	T r;
	cout << "Введите новый элемент массива: ";
	cin >> k;
	cout << "Введите номер элемента массива: ";
	cin >> r;
	for (int i = n - 1; i >= r - 1; i--)
	{
		a[i + 1] = a[i];
	}
	a[r - 1] = k;
	n++;
	for (int i = 0; i < n; i++)
	{
		cout << a[i] << ' ';
	}
	cout << endl;
	return 0;
}
